% ============================================================

% ---- 禁用自动标题页，封面由 00-cover.md 提供 ----
\renewcommand{\maketitle}{}

% ---- 颜色定义 ----
\usepackage{xcolor}
\definecolor{bookred}{HTML}{722F37}
\definecolor{codebg}{HTML}{F5F5F0}
\definecolor{quotebar}{HTML}{722F37}

% ---- 章节标题样式 ----
\usepackage{titlesec}

% 章标题：大号加粗 + 下方装饰线
\titleformat{\chapter}[display]
  {\normalfont\huge\bfseries}
  {\chaptertitlename\ \thechapter}
  {20pt}
  {\Huge}
  [\vspace{2pt}{\color{bookred}\titlerule[1pt]}]

\titlespacing*{\chapter}{0pt}{-20pt}{30pt}

% 节标题：酒红色
\titleformat{\section}
  {\normalfont\Large\bfseries\color{bookred}}
  {\thesection}{1em}{}
\titlespacing*{\section}{0pt}{24pt}{12pt}

% 小节标题
\titleformat{\subsection}
  {\normalfont\large\bfseries}
  {\thesubsection}{1em}{}
\titlespacing*{\subsection}{0pt}{18pt}{8pt}

% ---- 代码块美化（使用 Pandoc 默认的 Shaded 环境）----
% Pandoc 生成 Shaded 环境用于代码块，这里重定义背景色
\usepackage{framed}
\definecolor{shadecolor}{HTML}{F5F5F0}

% ---- 页眉页脚 ----
\usepackage{fancyhdr}
\pagestyle{fancy}
\fancyhf{}
\fancyhead[L]{\small\textcolor{gray}{OpenClaw 零基础入门}}
\fancyhead[R]{\small\textcolor{gray}{\leftmark}}
\fancyfoot[C]{\small\textcolor{bookred}{\thepage}}
\renewcommand{\headrulewidth}{0.4pt}
\renewcommand{\footrulewidth}{0pt}

% 页眉线颜色
\renewcommand{\headrule}{{\color{bookred}\hrule width\headwidth height\headrulewidth\vskip-\headrulewidth}}

% 章首页无页眉
\fancypagestyle{plain}{
  \fancyhf{}
  \fancyfoot[C]{\small\textcolor{bookred}{\thepage}}
  \renewcommand{\headrulewidth}{0pt}
}

% ---- 目录 ----
\renewcommand{\contentsname}{目\quad 录}
\setcounter{tocdepth}{1}

% ---- 段落设置 ----
\setlength{\parindent}{0em}
\setlength{\parskip}{0.6em}

% ---- 列表设置 ----
\usepackage{enumitem}
\setlist[itemize]{nosep,leftmargin=1.5em,itemsep=3pt}
\setlist[enumerate]{nosep,leftmargin=1.5em,itemsep=3pt}

% ---- 表格 ----
\usepackage{booktabs}
